CREATE OR REPLACE TRIGGER CREATEBILL_INSERT
    AFTER INSERT ON "SYSTEM".ORDERS
    FOR EACH ROW
    BEGIN
  
        INSERT INTO "SYSTEM".BILLINGTABLE(ORDER_ID,FIRST_NAME,LAST_NAME,DISTRICT,NOTE)
    VALUES(:NEW.ORDER_ID,:NEW.FIRST_NAME,:NEW.LAST_NAME,:NEW.DISTRICT,'INSERTED');
  
   END;
 /

CREATE OR REPLACE TRIGGER CREATEBILL_UPDATE
    AFTER UPDATE ON "SYSTEM".ORDERS
    FOR EACH ROW
    BEGIN
  
        UPDATE "SYSTEM".BILLINGTABLE 
    SET ORDER_ID=:NEW.ORDER_ID,FIRST_NAME=:NEW.FIRST_NAME,LAST_NAME=:NEW.LAST_NAME,DISTRICT=:NEW.DISTRICT,NOTE='UPDATED' 
    WHERE ORDER_ID=:NEW.ORDER_ID;
  
   END;
 /
